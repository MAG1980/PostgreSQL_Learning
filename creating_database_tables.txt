    товар (id, id категории,  id производителя, артикул, название (модель), описание, количество, цена, скидка, дата создания),
CREATE TABLE products (
    id SERIAL PRIMARY KEY,
    category_id INT,
    manufacturer_id INT,
    vendor_code VARCHAR(50) NOT NULL UNIQUE,
    name VARCHAR(50) NOT NULL,
    description VARCHAR(150) NOT NULL,
    amount INT,
    price NUMERIC,
    discount NUMERIC,
    created_at TIMESTAMP
);

    категория товара (id категории, название категории, дата создания, id родительской категории),
CREATE TABLE product_categories(
    id SERIAL PRIMARY KEY,
    name VARCHAR(50),
    created_at TIMESTAMP
    parent_id INT
);

    производитель товара (id, название, id логотипа, страна происхождения, дата создания),
CREATE TABLE manufacturer (
    id SERIAL PRIMARY KEY,
    name VARCHAR(30) NOT NULL UNIQUE,
    brand_logo_id INT,
    country VARCHAR(30) NOT NULL,
    created_at TIMESTAMP
);

    логотип производителя товара (id, url логотипа, размер, дата создание),
CREATE TABLE brand_logo (
    id SERIAL PRIMARY KEY,
    url VARCHAR(250) NOT NULL UNIQUE,
    size INT NOT NULL,
    created_at TIMESTAMP
);

    фотография товара (id, url, размер, дата добавления),
CREATE TABLE goods_photos (
    id SERIAL PRIMARY KEY,
    url VARCHAR(250) NOT NULL UNIQUE,
    size INT NOT NULL,
    created_at TIMESTAMP
);

    отзыв о товаре (id товара, id пользователя, дата создания, рейтинг, содержание отзыва),
CREATE TABLE reviews (
    id SERIAL PRIMARY KEY,
    client_id INT NOT NULL UNIQUE,
    created_at TIMESTAMP,
    ratings_id INT,
    text_content VARCHAR(500);
);

    фотография к отзыву (id фотографии, id отзыва, url, размер, дата добавления),
CREATE TABLE clients_photos (
    id SERIAL PRIMARY KEY,
    reviews_id INT NOT NULL,
    url VARCHAR(250) NOT NULL UNIQUE,
    size INT NOT NULL,
    created_at TIMESTAMP
);

    клиент (id пользователя, id аватара, имя, фамилия, дата рождения, электронная почта, телефон, дата регистрации),
CREATE TABLE clients (
    id SERIAL PRIMARY KEY,
    avatars_id INT,
    name VARCHAR(50) NOT NULL,
    surname VARCHAR(50) NOT NULL,
    birthday DATE NOT NULL,
    email VARCHAR(120) NOT NULL UNIQUE,
    phone VARCHAR(15) UNIQUE,
    registration_date DATE NOT NULL,
);

    аватар клиента (id аватара, url аватара, размер, дата создания),
CREATE TABLE avatars (
    id SERIAL PRIMARY KEY,
    url VARCHAR(250) NOT NULL UNIQUE,
    size INT NOT NULL,
    created_at TIMESTAMP
);

    избранное (id, id клиента),
CREATE TABLE favorites (
    id SERIAL PRIMARY KEY,
    clients_id INT NOT NULL,
);

    корзина (id, id клиента),
CREATE TABLE carts (
    id SERIAL PRIMARY KEY,
    clients_id NOT NULL
);

    заказ (id, id клиента, id способа оплаты, id пункта выдачи, дата оформления, статус),
CREATE TABLE orders (
    id SERIAl PRIMARY KEY,
    clients_id INT NOT NULL,
    payment_methods_id INT NOT NULL,
    pick-up_points_id INT NOT NULL,
    created_at TIMESTAMP,
    execution_statuses_id INT NOT NULL
);

    способ оплаты (id, название, дата создания),
CREATE TABLE payment_methods (
    id SERIAL PRIMARY KEY,
    name VARCHAR(20),
    created_at TIMESTAMP
);

    статус заказа (id, название, дата создания),
CREATE TABLE execution_statuses (
    id SERIAL PRIMARY KEY,
    name VARCHAR(20),
    created_at TIMESTAMP
);

    пункт выдачи заказов (id, индекс, город, улица, дом, дата создания),
CREATE TABLE pick-up_points (
    id SERIAL PRIMARY KEY,
    postal_code VARCHAR(6),
    city VARCHAR(30),
    street VARCHAR(50),
    building VARCHAR(4),
    created_at TIMESTAMP,
);

    рейтинг (id, название, оценка),
CREATE TABLE ratings (
    id SERIAL PRIMARY KEY,
    name VARCHAR(10) NOT NULL,
    grade INT NOT NULL
);

    заказ_товар - (id заказа, id товара, количество товара, цена товара, скидка),
CREATE TABLE orders_products (
    orders_id NOT NULL,
    products_id INT NOT NULL,
    product_amount INT NOT NULL,
    product_price NUMERIC,
    discount NUMERIC
);

    корзина_товар - (id корзины, id товара, количество товара, цена товара, скидка),
CREATE TABLE carts_products (
    carts_id INT NOT NULL,
    products_id INT NOT NULL,
    product_amount INT NOT NULL,
    product_price NUMERIC,
    discount NUMERIC
);

    избранное_товар: (id избранного, id товара, дата добавления).
CREATE TABLE favorites_products (
    favorites_id INT NOT NULL,
    products_id INT NOT NULL,
    date_added DATE NOT NULL
);
